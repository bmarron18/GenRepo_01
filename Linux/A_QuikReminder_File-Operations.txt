	sudo cp -r fontawesome5 /usr/local/share/texmf/tex/latex/
	cd /usr/local/share/texmf/tex/latex/fontawesome5/
	sudo mktexlsr
	
	cd ~/Desktop/LaTex/LaTex_installed_packages
	sudo cp -r moderncv-old /usr/local/share/texmf/tex/latex/
	cd /usr/local/share/texmf/tex/latex
	sudo mv moderncv-old moderncv
	cd /usr/local/share/texmf/tex/latex/moderncv/
	sudo mktexlsr
	
COPY just (all) files and hidden folders
	cp -a /source/. /dest/ 	<== -a option is an improved recursive option
							<== The . at end of the source path is a specific cp syntax that copies 							all files and folders, including hidden ones.
	
MOVE entire directory w/ all files!!
